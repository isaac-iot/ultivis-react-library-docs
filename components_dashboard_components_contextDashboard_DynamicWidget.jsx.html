import { memo, useState } from 'react';
import { widgetManager } from '@utils';
import { useTranslation } from '@hooks';

/**
 * A dynamic widget loader that displays widgets based on their type and ID.
 * Supports rendering a preview image with search query highlighting or a widget component.
 *
 * @param {Object} props - The component props.
 * @param {string} props.widgetId - The ID of the widget to load.
 * @param {string} [props.searchQuery] - The search query for highlighting matches in the widget label.
 * @param {string} props.type - The type of widget to render (e.g., 'previewImage', 'configComponent').
 * @param {function} [props.onSelect] - Callback function triggered when the widget is selected.
 * @param {Object} [props.props] - Additional props to pass to the widget component.
 *
 * @returns {JSX.Element|null} The rendered widget or a preview image with highlighted search results.
 *
 * @example
 * <DynamicWidget
 *   widgetId="widget123"
 *   searchQuery="example"
 *   type="previewImage"
 *   onSelect={(widget) => console.log('Selected widget:', widget)}
 * />
 */
const DynamicWidget = ({ widgetId, searchQuery, type, onSelect, ...props }) => {
  const { t } = useTranslation();
  const [error, setError] = useState('');
  const widgetComponent = widgetManager.load(widgetId);

  if (error) {
    return (
      <div>
        {t('Error loading widget:')} {error}
      </div>
    );
  }

  if (type === 'previewImage') {
    const path = widgetComponent && widgetComponent['previewImage'];
    const label = widgetComponent && widgetComponent['label'];

    const highlightedLabel = (label) => {
      if (!searchQuery) return t(label);

      const regex = new RegExp(`(${searchQuery})`, 'gi');
      const searchQueries = t(label).split(regex);

      return searchQueries.map((query, index) =>
        regex.test(query) ? (
          <span
            key={index}
            className="font-bold text-primary dark:text-dark-primary"
          >
            {query}
          </span>
        ) : (
          <span key={index}>{query}</span>
        )
      );
    };

    return (
      <div
        className="rounded-lg bg-grayscale-900 p-1 duration-150 hover:ring-4 hover:ring-blue-500 dark:bg-dark-grayscale-900"
        {...props}
        onClick={() => onSelect(widgetComponent)}
      >
        <div className="rounded-lg bg-grayscale-1000 p-2 dark:bg-dark-bg-333">
          <div className="bg-gray-200 mb-2 w-full p-2">
            <img src={path} alt={t(label)} className="aspect-square" />
          </div>
          <div className="text-center text-sm uppercase">
            {highlightedLabel(label)}
          </div>
        </div>
      </div>
    );
  }

  const Widget = widgetComponent && widgetComponent[type];
  return Widget ? <Widget {...props} /> : null;
};

export default memo(DynamicWidget);
