import { useEffect } from 'react';
import UAParser from 'ua-parser-js';
import { useSidebarStore } from '@layouts';
import { useUserAgentStore } from '@stores';
import DashboardActionItems from '../components/contextDashboard/ConetextDashboardActionItems';
import { useDashboardStore } from '../store/useDashboardStore';

/**
 * A responsive and dynamic action bar component that adjusts its behavior based on the sidebar state,
 * device type, and dashboard status. It serves as a container for action items and custom child elements.
 *
 * @param {Object} props - The component props.
 * @param {React.ReactNode} props.children - Optional child elements to render within the action bar.
 *
 * @returns {JSX.Element} A fully responsive action bar component.
 *
 * @example
 * <ActionBar>
 *   <CustomActionItem />
 * </ActionBar>
 *
 * @description
 * - The `ActionBar` component adjusts its position and visibility based on whether the header is open (`headerIsOpen`)
 *   or the current device type (mobile, wearable, or desktop) derived from the user agent.
 * - If `isDashboard` is true, the `DashboardActionItems` component is rendered as part of the action bar.
 * - It uses `UAParser` to determine the device type and updates the user agent store accordingly.
 */
const ActionBar = ({ children }) => {
  const { headerIsOpen } = useSidebarStore((state) => state);
  const isDevice = useUserAgentStore((state) => state.isDevice); // mobile, wearable 체크
  const { isDashboard } = useDashboardStore((state) => state);

  useEffect(() => {
    if (isDevice === null) {
      const parser = new UAParser();
      const { type } = parser.getDevice();
      const setIsDevice = useUserAgentStore.getState().setIsDevice;
      setIsDevice(type);
    }
  }, []);

  return (
    <>
      {!isDevice && (
        <div
          className={`ease fixed left-0 w-full transition-[top] duration-150 sm:static ${
            headerIsOpen ? 'top-32' : 'top-16'
          } z-[40]`}
        >
          <ul
            className={`flex w-full ${
              !headerIsOpen
                ? 'h-0 w-0 divide-y-[0px] py-0 shadow-none'
                : 'h-[calc(100vh-8rem)] divide-y-[1px] py-1 shadow-lg '
            } flex-col overflow-hidden whitespace-nowrap bg-white text-base text-grayscale-400 shadow-lg dark:bg-dark-bg-444 sm:relative sm:top-0 sm:z-0 sm:h-11.5 sm:flex-row sm:justify-end sm:gap-x-4 sm:divide-y-0 sm:bg-transparent sm:shadow-none  sm:dark:bg-transparent`}
          >
            {isDashboard && <DashboardActionItems />}
            {children}
          </ul>
        </div>
      )}
    </>
  );
};

export default ActionBar;
