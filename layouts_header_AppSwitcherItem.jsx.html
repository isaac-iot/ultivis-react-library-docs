import { NavigationMenuLink } from '@components';
import { humanizeAppName } from '@utils';
import {
  AdvancedSimulatorsIcon,
  AdministrationIcon,
  CockpitIcon,
  MigrationToolIcon,
  MachineLearningWorkbenchIcon,
  DeviceManagementIcon,
} from '@styles';
import { useTranslation } from '@hooks';

const getIconComponent = (iconName) => {
  const icons = {
    administration: AdministrationIcon,
    simulators: AdvancedSimulatorsIcon,
    cockpit: CockpitIcon,
    devicemanagement: DeviceManagementIcon,
    'migration-tool': MigrationToolIcon,
    'machine-learning-workbench': MachineLearningWorkbenchIcon,
  };

  return icons[iconName] || null; // iconName이 없으면 null 반환
};

const getInitials = (name) => {
  const firstLetter = name.slice(0, 1).toUpperCase();
  const secondLetter = name.slice(1, 2).toLowerCase();
  return firstLetter + secondLetter;
};

/**
 * AppSwitcherItem Component
 *
 * Renders a single application item with an icon and label, providing navigation to the app's URL.
 *
 * @component
 * @param {Object} props - Component properties.
 * @param {Object} props.app - Application data object.
 * @param {string} props.app.contextPath - Context path for the application.
 * @param {string} props.app.externalUrl - External URL for the application.
 * @param {string} props.app.name - Name of the application.
 * @param {Object} [props.app.manifest] - Manifest object containing metadata for the application.
 * @param {Object} [props.app.manifest.icon] - Icon metadata for the application.
 * @param {string} [props.app.manifest.icon.class] - CSS class for the application icon.
 */
const AppSwitcherItem = ({ app }) => {
  const { t } = useTranslation();

  const href = app.contextPath ? `/apps/${app.contextPath}/` : app.externalUrl;
  const IconComponent = getIconComponent(app.contextPath);

  return (
    <div className="group rounded-2xl border-2 border-grayscale-700 bg-white hover:bg-primary/10 hover:text-primary dark:border-dark-bg-333 dark:border-dark-grayscale-700 dark:bg-dark-bg-444 dark:text-dark-grayscale-200 dark:hover:bg-dark-primary/10 dark:hover:text-dark-primary">
      <li className="h-full">
        <NavigationMenuLink asChild>
          <a href={href}>
            <div className="my-auto flex h-full w-full transform flex-col items-center space-y-1.5 px-2 pb-2 pt-3 group-hover:text-primary dark:group-hover:text-dark-primary">
              {/* 커스텀 SVG 아이콘이 존재하는 경우 */}
              {IconComponent ? (
                <IconComponent className="h-9 min-h-[36px] w-9 min-w-[36px] group-hover:text-primary dark:group-hover:text-dark-primary" />
              ) : /* 커스텀 아이콘이 없고, app.manifest.icon.class 속성이 있는 경우 */
              app?.manifest?.icon && app.manifest.icon.class ? (
                <div
                  className={`c8y-icon ${app.manifest.icon.class} text-[36px] group-hover:text-primary`}
                ></div>
              ) : (
                /* 위의 조건을 만족하지 않으면 getInitials 실행 */
                <div className="flex h-9 w-9 items-center justify-center rounded-lg border-[3px] border-grayscale-100 text-lg font-bold text-grayscale-100 group-hover:border-primary group-hover:text-primary dark:border-dark-grayscale-100 dark:text-dark-grayscale-100  dark:group-hover:border-dark-primary  dark:group-hover:text-dark-primary">
                  {getInitials(app.name)}
                </div>
              )}
              <span className="flex h-full max-w-full items-center whitespace-normal break-words text-center text-xs font-medium text-grayscale-100 group-hover:text-primary dark:text-dark-grayscale-100 dark:group-hover:text-dark-primary">
                {t(humanizeAppName(app.name))}
              </span>
            </div>
          </a>
        </NavigationMenuLink>
      </li>
    </div>
  );
};

export default AppSwitcherItem;
