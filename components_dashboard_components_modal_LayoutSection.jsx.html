import { useFormContext } from 'react-hook-form';
import {
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  Input,
  Label,
  RadioGroup,
  RadioGroupItem,
  QuestionTooltip,
} from '@components';
import { cn } from '@utils';
import { useTranslation } from '@hooks';

/**
 * A layout configuration section for setting widget header styles and margins.
 *
 * @param {Object} props - The component props.
 * @param {'add'|'edit'|'widget'} props.mode - The mode of the configuration, such as 'add', 'edit', or 'widget'.
 * @param {string} [props.className] - Additional class names for custom styling.
 * @param {string} props.headerStyle - The default style of the widget header (e.g., 'panel-title-regular').
 *
 * @returns {JSX.Element} A layout configuration section for the dashboard.
 *
 * @example
 * <LayoutSection mode="add" className="custom-class" headerStyle="panel-title-regular" />
 */
const LayoutSection = ({ mode, className, headerStyle }) => {
  const { t } = useTranslation();
  const { control } = useFormContext();

  return (
    <div className={cn('flex flex-col gap-y-6', className)}>
      <div className="flex items-center gap-x-2">
        <Label className="min-w-fit text-lg font-bold">{t('LAYOUT')}</Label>
        {/* <Separator className="h-[2px] flex-1" /> */}
      </div>
      <FormField
        className="flex gap-x-6"
        name="headerStyle"
        control={control}
        render={({ field: { onChange } }) => (
          <FormItem className="space-y-2">
            <FormLabel>{t('Widget header style')}</FormLabel>
            <FormControl>
              <RadioGroup
                className="gap-0 rounded-lg border border-grayscale-600 dark:border-dark-grayscale-600"
                onValueChange={onChange}
                defaultValue={headerStyle}
              >
                <FormItem className="flex items-center gap-x-2 border-b-[1px] border-grayscale-600 px-4 py-3 dark:border-dark-grayscale-600">
                  <FormControl>
                    <RadioGroupItem
                      value="panel-title-regular"
                      id="panel-title-regular"
                    />
                  </FormControl>
                  <div className="flex w-full items-center justify-between">
                    <FormLabel htmlFor="panel-title-regular">
                      {t('Regular')}
                    </FormLabel>
                    <QuestionTooltip
                      description={t(
                        'The widget has no border between header and content.'
                      )}
                    />
                  </div>
                </FormItem>
                <FormItem className="flex items-center gap-x-2 border-b-[1px] border-grayscale-600 px-4 py-3 dark:border-dark-grayscale-600">
                  <FormControl>
                    <RadioGroupItem
                      value="panel-title-border"
                      id="panel-title-border"
                    />
                  </FormControl>
                  <div className="flex w-full items-center justify-between">
                    <FormLabel htmlFor="panel-title-border">
                      {t('Border')}
                    </FormLabel>
                    <QuestionTooltip
                      description={t(
                        'The widget has a small separation border between header and content.'
                      )}
                    />
                  </div>
                </FormItem>
                <FormItem className="flex items-center gap-x-2 px-4 py-3">
                  <FormControl>
                    <RadioGroupItem
                      value="panel-title-hidden"
                      id="panel-title-hidden"
                    />
                  </FormControl>
                  <div className="flex w-full items-center justify-between">
                    <FormLabel htmlFor="panel-title-hidden">
                      {t('Hidden')}
                    </FormLabel>
                    <QuestionTooltip
                      description={t('The widget header is not shown.')}
                    />
                  </div>
                </FormItem>
              </RadioGroup>
            </FormControl>
          </FormItem>
        )}
      />

      {mode !== 'widget' && (
        <FormField
          className="flex gap-x-6"
          control={control}
          name="widgetMargin"
          rules={{ min: 0, max: 50 }}
          render={({ field }) => (
            <FormItem className="space-y-2">
              <FormLabel>{t('Widget margin')}</FormLabel>
              <FormControl>
                <div className="flex w-full items-center rounded-lg border border-grayscale-600 bg-grayscale-1000 dark:border-dark-grayscale-600 dark:bg-dark-bg-444">
                  <Input
                    className="right w-full rounded-md border-r-0 border-none px-2 leading-6"
                    type="number"
                    {...field}
                  />
                  <div className="mr-4 h-6 w-6 bg-grayscale-1000 dark:bg-dark-bg-444">
                    px
                  </div>
                </div>
              </FormControl>
            </FormItem>
          )}
        />
      )}
    </div>
  );
};

export default LayoutSection;
