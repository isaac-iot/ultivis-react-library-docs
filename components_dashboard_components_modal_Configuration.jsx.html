import { useEffect } from 'react';
import { useFormContext, useWatch } from 'react-hook-form';
import { FocusScope } from '@radix-ui/react-focus-scope';
import {
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
  Input,
  Separator,
  SearchDeviceTree,
  DataManager,
} from '@components';
import { useTranslation, useDeviceInventory } from '@hooks';
import DynamicWidget from '../contextDashboard/DynamicWidget';

/**
 * A configuration component for managing the setup of a selected widget.
 * Provides forms and dynamic UI elements for setting the title, device, and widget configuration.
 *
 * @param {Object} props - The component props.
 * @param {Object} props.selectedWidget - The selected widget object containing configuration details.
 * @param {string} props.sourceId - The ID of the source associated with the widget.
 *
 * @returns {JSX.Element} A configuration UI for the selected widget.
 *
 * @example
 * <Configuration selectedWidget={widget} sourceId="123456" />
 */
const Configuration = ({ selectedWidget, sourceId }) => {
  const { t } = useTranslation();
  const { control, setValue, getValues } = useFormContext();

  const { data: device } = useDeviceInventory(sourceId, {
    enabled: !selectedWidget.data.options.noDeviceTarget || !!sourceId,
    refetchOnMount: 'always',
    select: (data) => (data?.data ? data.data : data),
  });

  const link = useWatch({ name: 'link' });

  useEffect(() => {
    if (!device) return;
    const selectedDevice = getValues('config.device');
    if (!selectedDevice) return;
    // CHECKLIST : 기존 위젯 때문에 나중에 수정 selectedDevice.id ?? device.id
    const updateDeviceId =
      selectedDevice.id === '' || !selectedDevice.id
        ? device.id
        : selectedDevice.id;
    const updateDeviceName =
      selectedDevice.name === '' || !selectedDevice.name
        ? device.name
        : selectedDevice.name;
    if (sourceId)
      setValue('config.device', {
        id: updateDeviceId,
        name: updateDeviceName,
      });
  }, [device, sourceId]);

  return (
    <>
      <FocusScope>
        {selectedWidget && (
          <div className="flex w-full flex-col justify-center pb-3">
            <div className="py-8">
              <div>
                <span className="text-lg font-bold text-grayscale-200 dark:text-dark-grayscale-200">
                  {t(selectedWidget.label)}
                </span>
                <p className="text-sm font-medium text-grayscale-300 dark:text-dark-grayscale-300">
                  {t(selectedWidget.description)}
                </p>
              </div>
            </div>
            {/* 위젯 제목 */}
            <div>
              <FormField
                control={control}
                name="title"
                render={({ field }) => (
                  <FormItem className="mb-6 space-y-1.5">
                    <FormLabel>{t('Title')}</FormLabel>
                    <FormControl>
                      <Input
                        className="text-base font-bold"
                        placeholder={selectedWidget.label}
                        {...field}
                        value={field.value}
                        clearable={true}
                      />
                    </FormControl>
                    <FormMessage />
                  </FormItem>
                )}
              />
              <DataManager dataType="link" name="link" data={link} />
            </div>

            {!selectedWidget.data.options.noDeviceTarget && (
              <div className="w-full">
                <FormField
                  name="config.device"
                  control={control}
                  render={({ field: { onChange } }) => (
                    <FormItem className="space-y-1.5">
                      <FormLabel>{t('Target assets or devices')}</FormLabel>
                      <FormControl>
                        <SearchDeviceTree
                          groupSelectable={
                            selectedWidget.data.options.groupsSelectable
                          }
                          context={device}
                          selected={(props) => {
                            if (!props) {
                              return;
                            }
                            const id = props.id ?? '';
                            const name = props.name ?? '';

                            onChange({ id, name });
                          }}
                          getSelectedDevice={
                            selectedWidget?.config?.device?.id || null
                          }
                        />
                      </FormControl>
                      <FormMessage />
                    </FormItem>
                  )}
                />

                <Separator className="my-6" />
              </div>
            )}
            <div className="flex flex-col">
              <DynamicWidget
                widgetId={selectedWidget['id']}
                type="configComponent"
              />
            </div>
          </div>
        )}
      </FocusScope>
    </>
  );
};

export default Configuration;
