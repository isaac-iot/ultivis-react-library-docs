/**
 * Encodes authentication credentials into a Base64 string.
 *
 * @function encodeAuth
 * @param {Object} params - The authentication parameters.
 * @param {string} params.tenant - The tenant identifier.
 * @param {string} params.user - The username.
 * @param {string} params.password - The password.
 * @returns {string} - A Base64 encoded string in the format `tenant/user:password`.
 *
 * @example
 * const authString = encodeAuth({ tenant: 'example', user: 'admin', password: '12345' });
 * console.log(authString); // Encoded Base64 string
 */
export const encodeAuth = ({ tenant, user, password }) =>
  btoa(`${tenant}/${user}:${password}`);

/**
 * Stores authentication credentials in the session storage using Base64 encoding.
 *
 * @function setBasicAuth
 * @param {Object} params - The authentication parameters.
 * @param {string} params.tenant - The tenant identifier.
 * @param {string} params.user - The username.
 * @param {string} params.password - The password.
 *
 * @description
 * The encoded credentials are stored in session storage under the key `_tcy8`.
 *
 * @example
 * setBasicAuth({ tenant: 'example', user: 'admin', password: '12345' });
 */
export const setBasicAuth = ({ tenant, user, password }) =>
  sessionStorage.setItem('_tcy8', encodeAuth({ tenant, user, password }));
