import { has } from 'lodash-es';

/**
 * Updates the device information in the dashboard configuration based on the device type.
 *
 * @function deviceTypeDashboardForDevice
 * @param {Object} config - The current dashboard configuration.
 * @param {Object} context - The context object containing device information.
 * @returns {Object} - The updated dashboard configuration.
 */
export const deviceTypeDashboardForDevice = (config, context) => {
  if (has(context, 'u5s_IsParent') || has(context, 'c8y_IsDevice')) {
    return {
      ...config,
      device: {
        id: context.id,
        name: context.name,
      },
    };
  } else if (has(context, 'u5s_IsDevice')) {
    const managedObject = context.deviceParents.references[0].managedObject;
    return {
      ...config,
      device: {
        name: managedObject.name,
        id: managedObject.id,
      },
    };
  }
  return config;
};

/**
 * Compares two datapoints to check if they are the same.
 *
 * @function isSameDatapoint
 * @param {Object} d1 - The first datapoint object.
 * @param {Object} d2 - The second datapoint object.
 * @returns {boolean} - `true` if the datapoints are the same, `false` otherwise.
 */
export const isSameDatapoint = (d1, d2) => {
  return (
    d1?.fragment === d2?.fragment &&
    d1?.series === d2?.series &&
    d1?.__target.id === d2?.__target.id
  );
};

/**
 * Filters and returns the active datapoints from a list.
 *
 * @function getActiveData
 * @param {Object[]} dataList - The list of datapoints.
 * @returns {Object[]} - The filtered list of active datapoints.
 */
export const getActiveData = (dataList) => {
  return dataList?.filter((data) => data.__active) ?? [];
};

/**
 * Retrieves datapoints from localStorage.
 *
 * @function getDatapointsFromLocal
 * @param {string} id - The identifier for the datapoints.
 * @param {string} category - The category of the datapoints.
 * @param {boolean} [withActive=false] - Whether to include active status.
 * @returns {Object[]} - The list of retrieved datapoints.
 */
export const getDatapointsFromLocal = (id, category, withActive = false) => {
  const datapoints = JSON.parse(
    localStorage.getItem(`datapoints_${category}_${id}`) || '[]'
  );

  return withActive
    ? datapoints
    : datapoints?.map((datapoint) => {
        datapoint.__active = false;
        return datapoint;
      });
};

/**
 * Saves datapoints to localStorage.
 *
 * @function setDatapointsToLocal
 * @param {string} id - The identifier for the datapoints.
 * @param {string} category - The category of the datapoints.
 * @param {Object[]} newDataPoints - The list of new datapoints to save.
 */
export const setDatapointsToLocal = (id, category, newDataPoints) => {
  const datapoints = newDataPoints.map((datapoint) => {
    if (Object.prototype.hasOwnProperty.call(datapoint, 'yAxisType')) {
      const { yAxisType, ...data } = datapoint;
      return yAxisType ? { yAxisType, ...data } : data;
    } else {
      const { ...data } = datapoint;
      return data;
    }
  });

  localStorage.setItem(
    `datapoints_${category}_${id}`,
    JSON.stringify(datapoints)
  );
};

/**
 * Retrieves types from localStorage.
 *
 * @function getTypesFromLocal
 * @param {string} id - The identifier for the types.
 * @param {string} category - The category of the types.
 * @param {string} dataType - The data type of the types.
 * @returns {Object[]} - The list of retrieved types.
 */
export const getTypesFromLocal = (id, category, dataType) => {
  const typeList = JSON.parse(
    localStorage.getItem(`${dataType}_${category}_${id}`) || '[]'
  )?.map((type) => {
    type.__active = false;
    return type;
  });

  return typeList;
};

/**

export const setTypesToLocal = (id, category, dataType, newTypeList) => {

  localStorage.setItem(
    `${dataType}_${category}_${id}`,

  );
};
