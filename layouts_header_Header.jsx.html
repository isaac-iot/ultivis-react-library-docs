import { NavLink } from 'react-router-dom';
import { NavigationMenu } from '@components';
import {
  Logo02Icon,
  MenuDrawerIcon,
  ArrowLeftIcon,
  ArrowDownIcon,
} from '@styles';
import { useSidebarStore } from '../sidebar/useSidebarStore';

/**
 * Header Component
 *
 * Renders the application header with navigation, user menu, and other utilities.
 *
 * @component
 * @param {Object} props - Component properties.
 * @param {string} props.pageTitle - The title displayed in the header.
 * @param {React.ReactNode} [props.headerItems] - Additional navigation items for the header.
 */

/**
 * ListItem Component
 *
 * Represents an individual list item in the dropdown menus.
 *
 * @component
 * @param {Object} props - Component properties.
 * @param {React.ComponentType} [props.icon] - Icon component to display in the list item.
 * @param {string} [props.title] - Title text displayed in the list item.
 * @param {React.ReactNode} [props.children] - Optional child elements for the list item.
 */
const Header = ({ pageTitle, headerItems }) => {
  const { toggleSidebar, sidebarIsOpen, headerIsOpen, toggleHeader } =
    useSidebarStore((state) => state);

  return (
    <>
      {/* shadow-lg drop-shadow-1 */}
      <header
        className={`fixed top-0  flex w-full flex-col border-b bg-grayscale-1000 shadow-2 dark:bg-dark-bg-444 dark:drop-shadow-none sm:flex-row ${
          headerIsOpen ? 'z-50 max-h-32 shadow-none' : 'z-50'
        }`}
      >
        <div
          className={`${
            headerIsOpen ? 'max-h-32' : 'max-h-16 overflow-hidden'
          } ease flex w-full flex-col justify-start px-4 transition-[max-height] duration-150 sm:flex-row sm:items-center sm:justify-between sm:overflow-visible sm:border-none lg:px-12`}
        >
          <div className="my-auto flex min-h-16 items-center gap-3 sm:gap-4">
            <button
              aria-controls="sidebar"
              onClick={toggleSidebar}
              className="stroke-grayscale-300 dark:stroke-dark-grayscale-300"
            >
              {sidebarIsOpen ? (
                <ArrowLeftIcon className="aspect-square w-6 shrink-0" />
              ) : (
                <MenuDrawerIcon className="aspect-square w-6 shrink-0" />
              )}
            </button>
            <NavLink
              to="/"
              className={`${sidebarIsOpen ? 'block' : 'hidden 2xl:block'}`}
            >
              <Logo02Icon className="h-6 w-fit" />
            </NavLink>
            <div
              className={`${
                sidebarIsOpen ? 'hidden' : 'block tablet:hidden'
              } line-clamp-1 w-fit whitespace-nowrap text-2xl font-bold text-grayscale-100 dark:text-dark-grayscale-200`}
            >
              {pageTitle}
            </div>
          </div>
          <ul
            className={`relative z-50 my-auto flex min-h-16 w-full justify-center gap-2 sm:border-none`}
          >
            <div className="flex flex-1 items-center  gap-3 whitespace-nowrap text-base text-grayscale-100 dark:text-dark-grayscale-300 max-sm:flex-wrap">
              <NavigationMenu
                className="w-full max-w-full  justify-evenly sm:justify-end"
                viewportClassNames="right-auto sm:-right-4 lg:-right-12"
              >
                {headerItems}
              </NavigationMenu>
            </div>
          </ul>
        </div>

        {/* 헤더 확장 토글 버튼 - 모바일 */}
        <div className={`absolute -bottom-3 !z-[50] w-full sm:hidden`}>
          <button
            className="mx-auto flex h-6 w-10 justify-center rounded-full bg-white  shadow-2 dark:bg-dark-bg-444"
            onClick={() => toggleHeader()}
          >
            <ArrowDownIcon
              className={`${
                headerIsOpen ? 'rotate-180' : 'rotate-0'
              } transition-all duration-150`}
            />
          </button>
        </div>
      </header>
    </>
  );
};

export default Header;
