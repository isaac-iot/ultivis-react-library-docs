import {
  Checkbox,
  Input,
  Label,
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
  Separator,
  QuestionTooltip,
} from '@components';
import { Controller, useFormContext, useWatch } from 'react-hook-form';
import { useTranslation } from '@hooks';

/**
 * A general configuration section for a dashboard, allowing the user to set menu labels,
 * navigation position, and global application settings.
 *
 * @param {Object} props - The component props.
 * @param {string} props.type - The type of the dashboard or device associated with it.
 * @param {string} props.category - The category of the dashboard (e.g., "device", "group").
 * @param {'add'|'edit'} props.mode - The mode of the configuration, either 'add' or 'edit'.
 *
 * @returns {JSX.Element} The general configuration section for the dashboard.
 *
 * @example
 * <GeneralSection type="device" category="group" mode="add" />
 */
const GeneralSection = ({ type, category, mode }) => {
  const { t } = useTranslation();
  const { register, control } = useFormContext();
  const icon = useWatch({ name: 'icon' });
  return (
    <div className="flex flex-col gap-y-6">
      <div className="flex items-center gap-x-2">
        <Label className="min-w-fit text-lg font-bold">{t('GENERAL')}</Label>
      </div>
      <div className="flex items-center">
        {/* <div className="space-y-2">
          <Label>{t('Icon')}</Label>
          <Controller
            name="icon"
            control={control}
            defaultValue={false}
            render={({ field }) => (
              <Select
                value={icon}
                className="w-[80px]"
                onValueChange={(value) => field.onChange(value)}
              >
                <SelectTrigger className="w-[80px] focus:ring-0 focus:ring-offset-0">
                  <SelectValue />
                </SelectTrigger>
                <SelectContent>
                  <div className="grid grid-cols-3">
                    {icons.icons.map((icon) => (
                      <SelectItem value={icon} key={icon}>
                        <span className={icon} />
                      </SelectItem>
                    ))}
                  </div>
                </SelectContent>
              </Select>
            )}
          />
        </div> */}
        <div className="w-full space-y-2">
          <div className="space-x-1">
            <Label>{t('Menu label')}</Label>
            {/* <QuestionTooltip description="Menu label to display in submenu when dashboard is attached" /> */}
            <QuestionTooltip
              description={t(
                'Menu label to display in submenu when dashboard is attached'
              )}
            />
          </div>

          <Input type="text" name="name" {...register('name')} />
        </div>
      </div>
      <div className="space-y-2">
        <div className="space-x-1">
          <Label>{t('Position in navigation')}</Label>
          <QuestionTooltip
            description={t('Position in navigation menu (0 ~ 10000)')}
          />
        </div>
        <Input
          min={0}
          max={10000}
          type="number"
          name="priority"
          {...register('priority')}
        />
      </div>

      {/* group면 안보인다. */}
      <div className="flex items-center gap-x-2 pb-6">
        {category !== 'group' && mode !== 'edit' && (
          <>
            <Controller
              name="global"
              control={control}
              defaultValue={false}
              render={({ field }) => (
                <Checkbox onCheckedChange={(value) => field.onChange(value)} />
              )}
            />

            <Label>
              {t('Apply dashboard to all devices of type', { type })}
            </Label>
          </>
        )}
      </div>
    </div>
  );
};

export default GeneralSection;
