import { useEffect, useRef, useState } from 'react';
import { useForm, useWatch, FormProvider } from 'react-hook-form';
import {
  Button,
  Dialog,
  DialogTrigger,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogFooter,
  DialogClose,
  DataManager,
} from '@components';
import { useTranslation } from '@hooks';

/**
 * `ComputedPropertyConfig`
 *
 * A component for configuring computed properties in a dialog modal.
 *
 * @component
 *
 * @param {string} type - The type of computed property (e.g., 'measurement', 'alarm', 'event').
 * @param {string} path - The unique path or identifier for the computed property.
 * @param {Array} selectedTarget - An array of currently selected target items for the computed property.
 * @param {Object} context - Additional context for the configuration, such as data or related properties.
 * @param {Function} getComputedRow - Callback function to pass the configured computed property details.
 */
const ComputedPropertyConfig = ({
  type,
  path,
  selectedTarget,
  context,
  getComputedRow,
}) => {
  const name = 'target';
  const methods = useForm({
    defaultValues: {
      target: [],
    },
  });

  const { t } = useTranslation();
  const { control, reset } = methods;
  const target = useWatch({ control, name });
  const isMounted = useRef(false);
  const [previousTarget, setPreviousTarget] = useState([]);
  const dataType = type === 'measurement' ? 'datapoints' : `${type}s`;

  const resetTarget = () => {
    reset({
      target: previousTarget, // 취소 버튼을 누르면 이전 상태로 되돌림
    });
  };

  // 모달이 열리면 현재 타겟을 저장
  const handleDialogOpen = () => {
    setPreviousTarget([...target]);
  };

  // 데이터 타입을 바꿀 때 데이터 리스트 초기화
  useEffect(() => {
    if (isMounted.current) {
      // 초기 마운트가 아닌 경우에만 초기화
      reset({ target: [] });
    }
  }, [dataType, reset]);

  useEffect(() => {
    if (!isMounted.current) {
      if (selectedTarget && selectedTarget.length > 0 && type) {
        reset({
          target: [...selectedTarget],
        });
      }
      isMounted.current = true; // 초기 마운트 완료
    }
  }, [selectedTarget, reset, type]);

  const handleSaveClick = () => {
    getComputedRow({
      type,
      detail: target,
    });
  };

  const multiSelectable = ['alarmsStatusCheck'].includes(path); // 배열에 포함되는 api 는 여러개의 datapoint 선택 가능

  return (
    <>
      <Dialog onOpenChange={(isOpen) => isOpen && handleDialogOpen()}>
        <DialogTrigger className="h-10 w-full bg-orange-200">
          <i className="dlt-c8y-icon-bolt"></i>&nbsp;
          <span>{path}</span>
        </DialogTrigger>
        <DialogContent
          closeButton={false}
          className="w-full gap-0 rounded-2xl p-6 sm:p-7"
        >
          <DialogHeader>
            <DialogTitle className="p-0 text-xl font-bold">
              {t('Computed property configuration')}
            </DialogTitle>
          </DialogHeader>

          <p className="my-6 text-center text-xl">
            <i className="dlt-c8y-icon-bolt"></i>&nbsp;
            {path}
          </p>

          {type && (
            <FormProvider {...methods}>
              <DataManager
                dataType={dataType}
                data={target}
                multiSelect={multiSelectable}
                name={name}
                context={context}
              />
            </FormProvider>
          )}

          <DialogFooter>
            <DialogClose asChild>
              <Button
                onClick={resetTarget}
                type="button"
                variant="destructive"
                className="rounded-lg"
              >
                {t('Cancel')}
              </Button>
            </DialogClose>
            <DialogClose asChild>
              <Button
                onClick={handleSaveClick}
                type="button"
                disabled={target?.length === 0}
                className="rounded-lg"
              >
                {t('Save')}
              </Button>
            </DialogClose>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    </>
  );
};

export default ComputedPropertyConfig;
